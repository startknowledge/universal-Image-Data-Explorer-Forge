    const host = url.hostname;
    const mapping = {
      yourRepo With Domian,
    };
    const repo = mapping[host];
    if (repo) {
      const newPath = `/docs/${repo}${url.pathname}`;
      return fetch(`https://your-pages-domain.pages.dev${newPath}`);
    }
    return new Response('Not Found', { status: 404 });
  }
